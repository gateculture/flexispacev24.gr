function askBot(q, type) { 
    const flow = document.getElementById('support-flow');
    const options = document.getElementById('support-options');
    
    if(!flow) return;

    // Κρύβουμε τις επιλογές για να ξεκινήσει το chat
    if(options) options.style.display = 'none';

    // Εμφάνιση ερώτησης χρήστη (Δεξιά)
    flow.innerHTML += `
        <div class="flex justify-end animate-slideUp mb-4">
            <div class="bg-purple-600 text-white p-4 rounded-2xl font-bold shadow-sm max-w-[80%] text-sm">
                ${q}
            </div>
        </div>`;
    
    flow.scrollTop = flow.scrollHeight;

    // Στοχευμένη απάντηση Bot (Αριστερά)
    setTimeout(() => {
        let response = "";
        
        if(type === 'payment') {
            response = "Για προβλήματα πληρωμής, βεβαιωθείτε ότι το Wallet σας έχει υπόλοιπο. Δεχόμαστε Apple Pay, Google Pay και κάρτες. Αν η χρέωση απέτυχε, το ποσό επιστρέφεται αυτόματα.";
        } else if(type === 'host') {
            response = "Για να γίνετε Host, πατήστε το κουμπί (+) στην κεντρική οθόνη. Θα χρειαστεί να ανεβάσετε φωτογραφίες του χώρου και έγγραφα ιδιοκτησίας για έγκριση.";
        } else {
            response = "Είμαι ο Flexi Bot και είμαι εδώ για να βοηθήσω! Μπορείτε να μας στείλετε email στο flexispace@gmail.com.";
        }

        flow.innerHTML += `
            <div class="flex justify-start animate-slideUp mb-4">
                <div class="chat-them bg-white border border-gray-100 p-4 rounded-2xl font-bold shadow-sm max-w-[80%] italic text-gray-800 text-sm">
                    ${response}
                </div>
            </div>
            <div class="flex justify-center mt-4">
                <button onclick="resetBot()" class="text-[10px] font-black uppercase text-purple-600 tracking-widest bg-purple-50 px-4 py-2 rounded-full active:scale-95 transition">Πισω στισ επιλογεσ</button>
            </div>`;
        
        flow.scrollTop = flow.scrollHeight;
    }, 800);
}
